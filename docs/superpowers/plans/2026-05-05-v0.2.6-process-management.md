# v0.2.6 Process Management Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build complete optional process management for core runtime and daemon logs: background run, foreground run, status metadata, and unified log reading.

**Architecture:** Extend existing runtime state and launch helpers instead of adding a supervisor. Keep daemon lifecycle code intact and add a small log-reading module used by the root `logs` command. Keep CLI changes in `src/main.cpp` because this repository already routes root commands there.

**Tech Stack:** C++17, std::filesystem, POSIX fork/exec/wait/kill, CLI11, nlohmann_json, CMake, ctest.

int runCoreRuntimeForeground(const std::string& binary, const std::vector<std::string>& coreArgs) {
    pid_t pid = fork();
    if (pid < 0) {
        std::cerr << "foreground runtime fork failed\n";
        return ExitError;
    }
    if (pid == 0) {
        std::vector<char*> argv;
        argv.reserve(coreArgs.size() + 2);
        argv.push_back(const_cast<char*>(binary.c_str()));
        for (const auto& arg : coreArgs) {
            argv.push_back(const_cast<char*>(arg.c_str()));
        }
        argv.push_back(nullptr);
        execv(binary.c_str(), argv.data());
        _exit(127);
    }

    int status = 0;
    if (waitpid(pid, &status, 0) < 0) {
        std::cerr << "foreground runtime wait failed\n";
        return ExitError;
    }
    if (WIFEXITED(status)) {
        return WEXITSTATUS(status);
    }
    if (WIFSIGNALED(status)) {
        return 128 + WTERMSIG(status);
    }
    return ExitError;
}
```

```cpp
#include "subcli/logs.hpp"

#include <chrono>
#include <deque>
#include <fstream>
#include <iostream>
#include <thread>

namespace subcli {

std::vector<std::string> readLogTailLines(const std::filesystem::path& path, int tailLines, std::string& error) {
    error.clear();
    if (tailLines < 0) {
        error = "tail line count must be non-negative";
        return {};
    }
    std::ifstream in(path);
    if (!in) {
        error = "log file does not exist or cannot be read: " + path.string();
        return {};
    }

    std::vector<std::string> allLines;
    std::deque<std::string> tail;
    std::string line;
    while (std::getline(in, line)) {
        if (tailLines == 0) {
            allLines.push_back(line);
        } else {
            tail.push_back(line);
            if (static_cast<int>(tail.size()) > tailLines) {
                tail.pop_front();
            }
        }
    }

    if (tailLines == 0) {
        return allLines;
    }
    return std::vector<std::string>(tail.begin(), tail.end());
}

bool printLogTail(const std::filesystem::path& path, int tailLines, std::string& error) {
    const auto lines = readLogTailLines(path, tailLines, error);
    if (!error.empty()) {
        return false;
    }
    for (const auto& line : lines) {
        std::cout << line << '\n';
    }
    return true;
}

bool followLog(const std::filesystem::path& path, std::string& error) {
    error.clear();
    std::ifstream in(path, std::ios::binary);
    if (!in) {
        error = "log file does not exist or cannot be read: " + path.string();
        return false;
    }
    in.seekg(0, std::ios::end);
    while (true) {
        std::string chunk;
        std::getline(in, chunk);
        if (in) {
            std::cout << chunk << '\n';
            continue;
        }
        in.clear();
        std::this_thread::sleep_for(std::chrono::milliseconds(250));
    }
}

} // namespace subcli
```

```cpp
bool isRuntimeTargetName(const std::string& targetName) {
    ExportTarget target;
    std::string defaultFile;
    return parseExportTarget(targetName, target, defaultFile);
}

std::filesystem::path resolveLogsPath(const std::string& targetName, const std::string& fileOpt, std::string& error) {
    error.clear();
    if (!fileOpt.empty()) {
        return resolveFromCliCwd(fileOpt);
    }
    if (targetName == "daemon") {
        const auto status = inspectDaemonProcess(gPaths.stateDir, error);
        if (!error.empty()) {
            return {};
        }
        if (!status.options.logFile.empty()) {
            return status.options.logFile;
        }
        return daemonLogPath(gPaths.stateDir);
    }
    if (!isRuntimeTargetName(targetName)) {
        error = "unknown logs target: " + targetName;
        return {};
    }
    RuntimeStatus status;
    const auto statePath = runtimeStatePathForTarget(gPaths.stateDir, targetName);
    if (!loadRuntimeState(statePath, status, error)) {
        return {};
    }
    if (!status.logPath.empty()) {
        return status.logPath;
    }
    return runtimeLogPathForTarget(gPaths.stateDir, targetName);
}

int doLogsCommand(const std::vector<std::string>& args) {
    if (hasHelp(args)) {
        printLogsUsage();
        return ExitOk;
    }
    CLI::App parser("logs");
    parser.set_help_flag("");
    parser.allow_extras(false);
    std::string targetName;
    std::string fileOpt;
    int tailLines = 200;
    bool follow = false;
    parser.add_option("target", targetName);
    parser.add_option("--tail", tailLines);
    parser.add_flag("--follow", follow);
    parser.add_option("--file", fileOpt);
    if (!parseCliArgs(parser, args) || targetName.empty()) {
        printLogsUsage();
        return ExitUsage;
    }
    if (tailLines < 0) {
        std::cerr << "--tail must be non-negative\n";
        return ExitUsage;
    }
    std::string error;
    const auto path = resolveLogsPath(targetName, fileOpt, error);
    if (!error.empty()) {
        std::cerr << error << "\n";
        return ExitUsage;
    }
    if (!printLogTail(path, tailLines, error)) {
        std::cerr << "logs failed: " << error << "\n";
        return ExitError;
    }
    if (follow && !followLog(path, error)) {
        std::cerr << "logs follow failed: " << error << "\n";
        return ExitError;
    }
    return ExitOk;
}

Runtime process management is optional. By default, `subcli run <target>` starts a managed background core process and stores state under the active workspace state directory.

subcli run sing-box

subcli run sing-box --foreground

Daemon logs use the same log command:

```bash
subcli daemon start --target sing-box --log-file /tmp/subcli-daemon.log
subcli logs daemon --tail 200

`subcli status` shows the managed pid, config path, log path, and start time when a core is running.
