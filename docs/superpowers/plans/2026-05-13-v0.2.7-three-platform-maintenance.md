```cpp
#include <algorithm>

namespace subcli {

std::string platformPathListSeparator() {
#ifdef _WIN32
    return ";";
#else
    return ":";
#endif
}

std::vector<std::string> executableNameCandidates(const std::string& name) {
#ifdef _WIN32
    std::vector<std::string> out;
    const std::filesystem::path input(name);
    if (input.has_extension()) {
        out.push_back(name);
        return out;
    }
    const char* pathextRaw = std::getenv("PATHEXT");
    std::vector<std::string> extensions = {".exe", ".cmd", ".bat", ".com"};
    if (pathextRaw && *pathextRaw) {
        extensions.clear();
        std::stringstream ss(pathextRaw);
        std::string item;
        while (std::getline(ss, item, ';')) {
            if (!item.empty()) {
                std::transform(item.begin(), item.end(), item.begin(), [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
                extensions.push_back(item);
            }
        }
    }
    out.push_back(name);
    for (const auto& ext : extensions) {
        out.push_back(name + ext);
    }
    return out;
#else
    return {name};
#endif
}

std::string findExecutableOnPath(const std::string& name) {
    const char* raw = std::getenv("PATH");
    if (!raw || !*raw) {
        return "";
    }
    std::stringstream ss(raw);
    std::string dir;
    while (std::getline(ss, dir, platformPathListSeparator()[0])) {
        if (dir.empty()) {
            continue;
        }
        for (const auto& candidateName : executableNameCandidates(name)) {
            const auto candidate = std::filesystem::path(dir) / candidateName;
            if (isExecutablePath(candidate.string())) {
                return candidate.string();
            }
        }
    }
    return "";
}

} // namespace subcli
```

```cpp
#include "subcli/platform.hpp"

#include <algorithm>
#include <cerrno>
#include <chrono>
#include <cstring>
#include <filesystem>
#include <thread>

#include <fcntl.h>
#include <signal.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <unistd.h>

#ifdef __APPLE__
#include <mach-o/dyld.h>
#endif

namespace subcli {
namespace {

std::vector<char*> makeArgv(const std::string& binaryPath, const std::vector<std::string>& args) {
    std::vector<char*> argv;
    argv.reserve(args.size() + 2);
    argv.push_back(const_cast<char*>(binaryPath.c_str()));
    for (const auto& arg : args) {
        argv.push_back(const_cast<char*>(arg.c_str()));
    }
    argv.push_back(nullptr);
    return argv;
}

int decodeWaitStatus(int status) {
    if (WIFEXITED(status)) {
        return WEXITSTATUS(status);
    }
    if (WIFSIGNALED(status)) {
        return 128 + WTERMSIG(status);
    }
    return -1;
}

} // namespace

std::string currentExecutablePath(const std::string& argv0) {
    std::error_code ec;
#if defined(__APPLE__)
    uint32_t size = 0;
    (void)_NSGetExecutablePath(nullptr, &size);
    std::string buffer(size, '\0');
    if (_NSGetExecutablePath(buffer.data(), &size) == 0) {
        buffer.resize(std::strlen(buffer.c_str()));
        return std::filesystem::absolute(buffer, ec).lexically_normal().string();
    }
#else
    const auto procSelf = std::filesystem::read_symlink("/proc/self/exe", ec);
    if (!ec && !procSelf.empty()) {
        return std::filesystem::absolute(procSelf, ec).lexically_normal().string();
    }
#endif
    if (!argv0.empty()) {
        return std::filesystem::absolute(argv0, ec).lexically_normal().string();
    }
    return "";
}

bool isExecutablePath(const std::string& path) {
    if (path.empty()) {
        return false;
    }
    return access(path.c_str(), X_OK) == 0;
}

ProcessRunResult runProcessCapture(const std::string& binaryPath, const std::vector<std::string>& args, int timeoutSec) {
    ProcessRunResult result;
    if (!isExecutablePath(binaryPath)) {
        result.error = "binary path is not executable: " + binaryPath;
        return result;
    }

    int pipeFd[2] = {-1, -1};
    if (pipe(pipeFd) != 0) {
        result.error = std::string("failed to create pipe: ") + std::strerror(errno);
        return result;
    }

    pid_t pid = fork();
    if (pid < 0) {
        close(pipeFd[0]);
        close(pipeFd[1]);
        result.error = std::string("failed to fork: ") + std::strerror(errno);
        return result;
    }

    if (pid == 0) {
        close(pipeFd[0]);
        dup2(pipeFd[1], STDOUT_FILENO);
        dup2(pipeFd[1], STDERR_FILENO);
        close(pipeFd[1]);
        auto argv = makeArgv(binaryPath, args);
        execv(binaryPath.c_str(), argv.data());
        _exit(127);
    }

    result.started = true;
    close(pipeFd[1]);
    fcntl(pipeFd[0], F_SETFL, fcntl(pipeFd[0], F_GETFL, 0) | O_NONBLOCK);

    char buf[1024];
    int status = 0;
    const auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds(std::max(1, timeoutSec));
    while (true) {
        while (true) {
            const ssize_t n = read(pipeFd[0], buf, sizeof(buf));
            if (n > 0) {
                result.output.append(buf, static_cast<size_t>(n));
                continue;
            }
            break;
        }

        const pid_t done = waitpid(pid, &status, WNOHANG);
        if (done == pid) {
            break;
        }
        if (done < 0) {
            close(pipeFd[0]);
            result.error = std::string("failed to wait process: ") + std::strerror(errno);
            result.exitCode = -1;
            return result;
        }
        if (std::chrono::steady_clock::now() >= deadline) {
            kill(pid, SIGTERM);
            std::this_thread::sleep_for(std::chrono::milliseconds(200));
            if (waitpid(pid, &status, WNOHANG) == 0) {
                kill(pid, SIGKILL);
                waitpid(pid, &status, 0);
            }
            close(pipeFd[0]);
            result.timedOut = true;
            result.exitCode = 124;
            result.error = "command timed out after " + std::to_string(std::max(1, timeoutSec)) + " seconds";
            return result;
        }
        std::this_thread::sleep_for(std::chrono::milliseconds(20));
    }

    while (true) {
        const ssize_t n = read(pipeFd[0], buf, sizeof(buf));
        if (n > 0) {
            result.output.append(buf, static_cast<size_t>(n));
            continue;
        }
        break;
    }
    close(pipeFd[0]);
    result.exitCode = decodeWaitStatus(status);
    if (result.exitCode != 0 && result.error.empty()) {
        result.error = result.output.empty() ? "command failed with exit code " + std::to_string(result.exitCode) : result.output;
    }
    return result;
}

int runProcessForeground(const std::string& binaryPath, const std::vector<std::string>& args, std::string& error) {
    error.clear();
    pid_t pid = fork();
    if (pid < 0) {
        error = "foreground runtime fork failed";
        return 1;
    }
    if (pid == 0) {
        auto argv = makeArgv(binaryPath, args);
        execv(binaryPath.c_str(), argv.data());
        _exit(127);
    }
    int status = 0;
    if (waitpid(pid, &status, 0) < 0) {
        error = "foreground runtime wait failed";
        return 1;
    }
    return decodeWaitStatus(status);
}

BackgroundProcessResult startBackgroundProcess(const std::string& binaryPath, const std::vector<std::string>& args, const std::filesystem::path& logPath) {
    BackgroundProcessResult result;
    if (!isExecutablePath(binaryPath)) {
        result.error = "binary path is not executable: " + binaryPath;
        return result;
    }
    std::error_code ec;
    if (!logPath.parent_path().empty()) {
        std::filesystem::create_directories(logPath.parent_path(), ec);
        if (ec) {
            result.error = "failed to create log directory: " + ec.message();
            return result;
        }
    }
    const int logFd = open(logPath.c_str(), O_WRONLY | O_CREAT | O_APPEND, 0644);
    if (logFd < 0) {
        result.error = "failed to open log file: " + logPath.string() + ": " + std::strerror(errno);
        return result;
    }

    pid_t pid = fork();
    if (pid < 0) {
        close(logFd);
        result.error = "failed to fork background process";
        return result;
    }
    if (pid == 0) {
        setsid();
        const int stdinFd = open("/dev/null", O_RDONLY);
        if (stdinFd >= 0) {
            dup2(stdinFd, STDIN_FILENO);
            close(stdinFd);
        }
        dup2(logFd, STDOUT_FILENO);
        dup2(logFd, STDERR_FILENO);
        close(logFd);
        auto argv = makeArgv(binaryPath, args);
        execv(binaryPath.c_str(), argv.data());
        _exit(127);
    }
    close(logFd);
    result.started = true;
    result.pid = static_cast<int>(pid);
    return result;
}

bool isProcessRunning(int pid) {
    if (pid <= 0) {
        return false;
    }
    if (kill(static_cast<pid_t>(pid), 0) == 0) {
        return true;
    }
    return errno != ESRCH;
}

bool terminateProcess(int pid, int timeoutSec, std::string& error) {
    error.clear();
    if (pid <= 0 || !isProcessRunning(pid)) {
        return true;
    }
    const pid_t posixPid = static_cast<pid_t>(pid);
    if (kill(posixPid, SIGTERM) != 0 && errno != ESRCH) {
        error = "failed to send SIGTERM";
        return false;
    }
    const auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds(std::max(1, timeoutSec));
    while (std::chrono::steady_clock::now() < deadline) {
        if (!isProcessRunning(pid)) {
            break;
        }
        std::this_thread::sleep_for(std::chrono::milliseconds(50));
    }
    if (isProcessRunning(pid)) {
        if (kill(posixPid, SIGKILL) != 0 && errno != ESRCH) {
            error = "failed to send SIGKILL";
            return false;
        }
    }
    int waitStatus = 0;
    (void)waitpid(posixPid, &waitStatus, WNOHANG);
    return true;
}

} // namespace subcli
```

```cpp
std::wstring utf8ToWide(const std::string& value) {
    if (value.empty()) {
        return L"";
    }
    const int size = MultiByteToWideChar(CP_UTF8, 0, value.c_str(), -1, nullptr, 0);
    std::wstring out(static_cast<size_t>(size - 1), L'\0');
    MultiByteToWideChar(CP_UTF8, 0, value.c_str(), -1, out.data(), size);
    return out;
}

std::string wideToUtf8(const std::wstring& value) {
    if (value.empty()) {
        return "";
    }
    const int size = WideCharToMultiByte(CP_UTF8, 0, value.c_str(), -1, nullptr, 0, nullptr, nullptr);
    std::string out(static_cast<size_t>(size - 1), '\0');
    WideCharToMultiByte(CP_UTF8, 0, value.c_str(), -1, out.data(), size, nullptr, nullptr);
    return out;
}

std::wstring quoteArgForCommandLine(const std::string& arg) {
    std::wstring input = utf8ToWide(arg);
    if (input.empty()) {
        return L"\"\"";
    }
    const bool needsQuotes = input.find_first_of(L" \t\n\v\"") != std::wstring::npos;
    if (!needsQuotes) {
        return input;
    }
    std::wstring out = L"\"";
    size_t backslashes = 0;
    for (wchar_t ch : input) {
        if (ch == L'\\') {
            ++backslashes;
            continue;
        }
        if (ch == L'\"') {
            out.append(backslashes * 2 + 1, L'\\');
            out.push_back(ch);
            backslashes = 0;
            continue;
        }
        out.append(backslashes, L'\\');
        backslashes = 0;
        out.push_back(ch);
    }
    out.append(backslashes * 2, L'\\');
    out.push_back(L'\"');
    return out;
}

std::wstring buildCommandLine(const std::string& binaryPath, const std::vector<std::string>& args) {
    std::wstring command = quoteArgForCommandLine(binaryPath);
    for (const auto& arg : args) {
        command.push_back(L' ');
        command += quoteArgForCommandLine(arg);
    }
    return command;
}

std::string lastWindowsError(const std::string& prefix) {
    const DWORD code = GetLastError();
    LPWSTR buffer = nullptr;
    FormatMessageW(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
                   nullptr,
                   code,
                   MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
                   reinterpret_cast<LPWSTR>(&buffer),
                   0,
                   nullptr);
    std::string message = prefix + ": " + std::to_string(code);
    if (buffer != nullptr) {
        message += " ";
        message += wideToUtf8(buffer);
        LocalFree(buffer);
    }
    return message;
}

std::string currentExecutablePath(const std::string& argv0) {
    std::wstring buffer(MAX_PATH, L'\0');
    DWORD len = GetModuleFileNameW(nullptr, buffer.data(), static_cast<DWORD>(buffer.size()));
    while (len == buffer.size() && GetLastError() == ERROR_INSUFFICIENT_BUFFER) {
        buffer.resize(buffer.size() * 2, L'\0');
        len = GetModuleFileNameW(nullptr, buffer.data(), static_cast<DWORD>(buffer.size()));
    }
    if (len > 0) {
        buffer.resize(len);
        return std::filesystem::absolute(std::filesystem::path(buffer)).lexically_normal().string();
    }
    return argv0.empty() ? "" : std::filesystem::absolute(argv0).lexically_normal().string();
}

bool isExecutablePath(const std::string& path) {
    std::error_code ec;
    const std::filesystem::path p(path);
    return !path.empty() && std::filesystem::is_regular_file(p, ec) && !ec;
}

ProcessRunResult runProcessCapture(const std::string& binaryPath, const std::vector<std::string>& args, int timeoutSec) {
    ProcessRunResult result;
    if (!isExecutablePath(binaryPath)) {
        result.error = "binary path is not executable: " + binaryPath;
        return result;
    }

    SECURITY_ATTRIBUTES sa{};
    sa.nLength = sizeof(sa);
    sa.bInheritHandle = TRUE;

    HANDLE readPipe = nullptr;
    HANDLE writePipe = nullptr;
    if (!CreatePipe(&readPipe, &writePipe, &sa, 0)) {
        result.error = lastWindowsError("failed to create pipe");
        return result;
    }
    SetHandleInformation(readPipe, HANDLE_FLAG_INHERIT, 0);

    STARTUPINFOW si{};
    si.cb = sizeof(si);
    si.dwFlags = STARTF_USESTDHANDLES;
    si.hStdInput = GetStdHandle(STD_INPUT_HANDLE);
    si.hStdOutput = writePipe;
    si.hStdError = writePipe;

    PROCESS_INFORMATION pi{};
    std::wstring command = buildCommandLine(binaryPath, args);
    std::vector<wchar_t> mutableCommand(command.begin(), command.end());
    mutableCommand.push_back(L'\0');

    const BOOL created = CreateProcessW(nullptr,
                                        mutableCommand.data(),
                                        nullptr,
                                        nullptr,
                                        TRUE,
                                        CREATE_NO_WINDOW,
                                        nullptr,
                                        nullptr,
                                        &si,
                                        &pi);
    CloseHandle(writePipe);
    if (!created) {
        CloseHandle(readPipe);
        result.error = lastWindowsError("failed to create process");
        return result;
    }

    result.started = true;
    std::thread reader([&]() {
        char buffer[1024];
        DWORD read = 0;
        while (ReadFile(readPipe, buffer, sizeof(buffer), &read, nullptr) && read > 0) {
            result.output.append(buffer, buffer + read);
        }
    });

    const DWORD waitMs = static_cast<DWORD>(std::max(1, timeoutSec)) * 1000U;
    const DWORD waitResult = WaitForSingleObject(pi.hProcess, waitMs);
    if (waitResult == WAIT_TIMEOUT) {
        TerminateProcess(pi.hProcess, 124);
        WaitForSingleObject(pi.hProcess, INFINITE);
        result.timedOut = true;
        result.exitCode = 124;
        result.error = "command timed out after " + std::to_string(std::max(1, timeoutSec)) + " seconds";
    } else if (waitResult != WAIT_OBJECT_0) {
        result.error = lastWindowsError("failed to wait process");
        result.exitCode = -1;
    } else {
        DWORD exitCode = 0;
        GetExitCodeProcess(pi.hProcess, &exitCode);
        result.exitCode = static_cast<int>(exitCode);
        if (result.exitCode != 0) {
            result.error = result.output.empty() ? "command failed with exit code " + std::to_string(result.exitCode) : result.output;
        }
    }

    CloseHandle(pi.hThread);
    CloseHandle(pi.hProcess);
    CloseHandle(readPipe);
    if (reader.joinable()) {
        reader.join();
    }
    return result;
}

int runProcessForeground(const std::string& binaryPath, const std::vector<std::string>& args, std::string& error) {
    error.clear();
    STARTUPINFOW si{};
    si.cb = sizeof(si);
    PROCESS_INFORMATION pi{};
    std::wstring command = buildCommandLine(binaryPath, args);
    std::vector<wchar_t> mutableCommand(command.begin(), command.end());
    mutableCommand.push_back(L'\0');
    const BOOL created = CreateProcessW(nullptr, mutableCommand.data(), nullptr, nullptr, TRUE, 0, nullptr, nullptr, &si, &pi);
    if (!created) {
        error = lastWindowsError("failed to create process");
        return 1;
    }
    WaitForSingleObject(pi.hProcess, INFINITE);
    DWORD exitCode = 1;
    GetExitCodeProcess(pi.hProcess, &exitCode);
    CloseHandle(pi.hThread);
    CloseHandle(pi.hProcess);
    return static_cast<int>(exitCode);
}

BackgroundProcessResult startBackgroundProcess(const std::string& binaryPath, const std::vector<std::string>& args, const std::filesystem::path& logPath) {
    BackgroundProcessResult result;
    if (!isExecutablePath(binaryPath)) {
        result.error = "binary path is not executable: " + binaryPath;
        return result;
    }
    std::error_code ec;
    if (!logPath.parent_path().empty()) {
        std::filesystem::create_directories(logPath.parent_path(), ec);
        if (ec) {
            result.error = "failed to create log directory: " + ec.message();
            return result;
        }
    }

    SECURITY_ATTRIBUTES sa{};
    sa.nLength = sizeof(sa);
    sa.bInheritHandle = TRUE;
    HANDLE logHandle = CreateFileW(utf8ToWide(logPath.string()).c_str(),
                                  FILE_APPEND_DATA,
                                  FILE_SHARE_READ | FILE_SHARE_WRITE,
                                  &sa,
                                  OPEN_ALWAYS,
                                  FILE_ATTRIBUTE_NORMAL,
                                  nullptr);
    if (logHandle == INVALID_HANDLE_VALUE) {
        result.error = lastWindowsError("failed to open log file");
        return result;
    }
    SetFilePointer(logHandle, 0, nullptr, FILE_END);

    STARTUPINFOW si{};
    si.cb = sizeof(si);
    si.dwFlags = STARTF_USESTDHANDLES;
    si.hStdInput = GetStdHandle(STD_INPUT_HANDLE);
    si.hStdOutput = logHandle;
    si.hStdError = logHandle;

    PROCESS_INFORMATION pi{};
    std::wstring command = buildCommandLine(binaryPath, args);
    std::vector<wchar_t> mutableCommand(command.begin(), command.end());
    mutableCommand.push_back(L'\0');
    const BOOL created = CreateProcessW(nullptr,
                                        mutableCommand.data(),
                                        nullptr,
                                        nullptr,
                                        TRUE,
                                        DETACHED_PROCESS | CREATE_NEW_PROCESS_GROUP,
                                        nullptr,
                                        nullptr,
                                        &si,
                                        &pi);
    CloseHandle(logHandle);
    if (!created) {
        result.error = lastWindowsError("failed to create background process");
        return result;
    }
    result.started = true;
    result.pid = static_cast<int>(pi.dwProcessId);
    CloseHandle(pi.hThread);
    CloseHandle(pi.hProcess);
    return result;
}

bool isProcessRunning(int pid) {
    if (pid <= 0) {
        return false;
    }
    HANDLE process = openProcessForStatus(pid, PROCESS_QUERY_LIMITED_INFORMATION | SYNCHRONIZE);
    if (process == nullptr) {
        return false;
    }
    DWORD exitCode = 0;
    const bool ok = GetExitCodeProcess(process, &exitCode) && exitCode == STILL_ACTIVE;
    CloseHandle(process);
    return ok;
}

bool terminateProcess(int pid, int timeoutSec, std::string& error) {
    error.clear();
    if (pid <= 0) {
        return true;
    }
    HANDLE process = OpenProcess(PROCESS_TERMINATE | SYNCHRONIZE | PROCESS_QUERY_LIMITED_INFORMATION, FALSE, static_cast<DWORD>(pid));
    if (process == nullptr) {
        return true;
    }
    DWORD exitCode = 0;
    if (GetExitCodeProcess(process, &exitCode) && exitCode != STILL_ACTIVE) {
        CloseHandle(process);
        return true;
    }
    if (!TerminateProcess(process, 1)) {
        error = lastWindowsError("failed to terminate process");
        CloseHandle(process);
        return false;
    }
    const DWORD waitMs = static_cast<DWORD>(std::max(1, timeoutSec)) * 1000U;
    const DWORD waitResult = WaitForSingleObject(process, waitMs);
    CloseHandle(process);
    if (waitResult != WAIT_OBJECT_0) {
        error = "failed to wait for terminated process";
        return false;
    }
    return true;
}
```

```cpp
bool validateRuntimeLogFileForCli(const std::string& logPath, std::string& error) {
    const std::filesystem::path path(logPath);
    const auto parent = path.parent_path();
    std::error_code ec;
    if (!parent.empty()) {
        std::filesystem::create_directories(parent, ec);
        if (ec) {
            error = "runtime log file is not writable: " + logPath;
            return false;
        }
    }

    std::ofstream out(logPath, std::ios::app);

        return false;
    }
    error.clear();
    return true;
}
```
